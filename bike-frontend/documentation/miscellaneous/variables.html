src/app/auth-interceptor.ts

authInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { // 1. Get the token const token = localStorage.getItem('token'); // 2. Clone the request if token exists if (token) { const cloned = req.clone({ setHeaders: { Authorization: `Bearer ${token}` } }); return next(cloned); } // 3. Pass through if no token return next(req); }

src/app/auth/role-guard.ts

roleGuard
Type : CanActivateFn
Default value : (route, state) => { const router = inject(Router); const token = localStorage.getItem('token'); if (!token) { router.navigate(['/login']); return false; } try { const payload = JSON.parse(atob(token.split('.')[1])); const userRole = payload.type; const allowedRoles = route.data['roles'] as Array<string>; if (allowedRoles.includes(userRole)) { return true; } else { router.navigate(['/']); return false; } } catch (e) { router.navigate(['/login']); return false; } }

Role-based route guard to restrict access based on user roles.
